Ejemplos de audio

Abrir el dispositivo de audio

    SDL_AudioSpec wanted;
    extern void fill_audio(void *udata, Uint8 *stream, int len);

    /* Set the audio format */
    wanted.freq = 22050;
    wanted.format = AUDIO_S16;
    wanted.channels = 2;    /* 1 = mono, 2 = stereo */
    wanted.samples = 1024;  /* Good low-latency value for callback */
    wanted.callback = fill_audio;
    wanted.userdata = NULL;

    /* Abrir el dispositivo de audio, forzando el formato deseado */
    if ( SDL_OpenAudio(&wanted, NULL) < 0 ) {
        fprintf(stderr, "Imposible abrir el audio: %s\n", SDL_GetError());
        return(-1);
    }
    return(0);

Reproducir audio

    static Uint8 *audio_chunk;
    static Uint32 audio_len;
    static Uint8 *audio_pos;

    /* La función de callback de audio toma los siguientes parametros:
       stream:  Un puntero al bufer que contendrá el audio (A pointer to the audio buffer to be filled)
       len:     La longitud (en bytes) del buffer de audio
    */
    void fill_audio(void *udata, Uint8 *stream, int len)
    {
        /* Solo reproducir si quedan datos */
        if ( audio_len == 0 )
            return;

        /* Mezclar tanta cantidad de datos como sea posible */
        len = ( len > audio_len ? audio_len : len );
        SDL_MixAudio(stream, audio_pos, len, SDL_MIX_MAXVOLUME)
        audio_pos += len;
        audio_len -= len;
    }

    /* Cargar datos de audio ... */

    audio_pos = audio_chunk;

    /* Let the callback function play the audio chunk */
    SDL_PauseAudio(0);

    /* Hacer algun proceso */

    ;;;;;

    /* Esperar a que el sonido termine */
    while ( audio_len > 0 ) {
        SDL_Delay(100);         /* Dormir 1/10 de segundo */
    }
    SDL_CloseAudio();
