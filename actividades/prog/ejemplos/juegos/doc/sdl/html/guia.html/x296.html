Ejemplos de CDROM

Listado de unidades de CD-ROM

    #include "SDL.h"

    /* Primero inicializar SDL */
    if ( SDL_Init(SDL_INIT_CDROM) < 0 ) {
        fprintf(stderr, "Imposible inicializar SDL: %s\n",SDL_GetError());
        exit(1);
    }
    atexit(SDL_Quit);

    /* Averiguar cuantas unidades de CD-ROM hay conectadas al sistema */
    printf("Unidades disponibles: %d\n", SDL_CDNumDrives());
    for ( i=0; i<SDL_CDNumDrives(); ++i ) {
        printf("Unidad %d:  \"%s\"\n", i, SDL_CDName(i));
    }

Abrir la unidad por defecto

    SDL_CD *cdrom;
    CDstatus status;
    char *status_str;

    cdrom = SDL_CDOpen(0);
    if ( cdrom == NULL ) {
        fprintf(stderr, "Imposible abrir la unidad de CD-ROM por defecto: %s\n",
                        SDL_GetError());
        exit(2);
    }

    status = SDL_CDStatus(cdrom);
    switch (status) {
        case CD_TRAYEMPTY:
            status_str = "bandeja vacia";
            break;
        case CD_STOPPED:
            status_str = "parado";
            break;
        case CD_PLAYING:
            status_str = "reproduciendo";
            break;
        case CD_PAUSED:
            status_str = "pausado";
            break;
        case CD_ERROR:
            status_str = "estado de error";
            break;
    }
    printf("El estado de la unidad: %s\n", status_str);
    if ( status >= CD_PLAYING ) {
        int m, s, f;
        FRAMES_TO_MSF(cdrom->cur_frame, &m, &s, &f);
        printf("Actuamente reproduciendo la pista %d, %d:%2.2d\n",
        cdrom->track[cdrom->cur_track].id, m, s);
    }

Listar las pistas de un CD

    SDL_CD *cdrom;          /* Asumiendo que ya ha sido inicializado.. */
    int i;
    int m, s, f;

    SDL_CDStatus(cdrom);
    printf("Pistas de la unidad: %d\n", cdrom->numtracks);
    for ( i=0; i<cdrom->numtracks; ++i ) {
        FRAMES_TO_MSF(cdrom->track[i].length, &m, &s, &f);
        if ( f > 0 )
            ++s;
        printf("\tPista (indice %d) %d: %d:%2.2d\n", i,
        cdrom->track[i].id, m, s);
    }

Reproducir un CD entero

    SDL_CD *cdrom;          /* Asumiendo que ya ha sido inicializado.. */

    // Reproducir el CD entero:
    if ( CD_INDRIVE(SDL_CDStatus(cdrom)) )
        SDL_CDPlayTracks(cdrom, 0, 0, 0, 0);

        // Reproducir la última pista:
        if ( CD_INDRIVE(SDL_CDStatus(cdrom)) ) {
            SDL_CDPlayTracks(cdrom, cdrom->numtracks-1, 0, 0, 0);
        }

        // Reproducir la primera y segunda pista y 10 segundos de la tercera:
        if ( CD_INDRIVE(SDL_CDStatus(cdrom)) )
            SDL_CDPlayTracks(cdrom, 0, 0, 2, 10);
